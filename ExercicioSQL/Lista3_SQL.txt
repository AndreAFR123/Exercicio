a.a)create table PECA(
	pec_codigo int(11) PRIMARY KEY auto_increment,
	pec_nome varchar(45) not null,
	pec_preco float(8,2) not null,
	pec_cor varchar(25) not null
);

a.b) create table UF (
	uf_codigo int(11) PRIMARY KEY auto_increment,
	uf_nome varchar(45) not null
);

a.c) create table CIDADE (
	cid_codigo int(11) PRIMARY KEY auto_increment,
	cid_nome varchar(45) not null,
	cid_uf int(11)
);

a.d) create table FORNECEDOR (
	for_codigo int(11) PRIMARY KEY auto_increment,
	for_nome varchar(45) not null,
	for_cidade int(11),
	for_categoria varchar(45) not null
);

a.e) create table PROJETO (
	pro_codigo int(11) PRIMARY KEY auto_increment,
	pro_nome varchar(45) not null,
	pro_duracao int(11) not null,
	pro_custo float(6,2) not null
);

a.f) create table FORNECE_PARA (
	fpa_peca int(11),
	fpa_fornecedor int(11),
	fpa_projeto int(11),
	fpa_qtd int(11)
);
